addInputString()

Description

Adds a control in the board. This control is a text input. When the focus is in the text input and the enter key is pressed, then the callback function is called with the string provided by the text field as an argument.

Syntax

webgui.addInputString("Name of the text field", fnAction)

Headers

int addInputString(const char * name, CallbackTypeCharp fnAction);

Parameters

name: The name of the text field

fnAction: Pointer to function that takes a char* as a parameter and returns void. This function will be called with a the string sent by the text input field from the interface, each time enter key is pressed.

Returns

The int id of the element

Example

#include <Webgui.h> 
#include <WiFi.h> 
  
Webgui webgui; //initialize an instance of the class 

char ssid[] = "Mywifi"; //  your network SSID (name)  
char pass[] = "0123ABCDEF";    // your network password (use for WPA, or use as key for WEP) 
int keyIndex = 0;// your network key Index number (needed only for WEP) 
char server[] = "192.168.1.17"; 
WiFiClient client; 
int status = WL_IDLE_STATUS; 

void actionString(char * value){ 
    Serial.print("Your name is: "); 
    Serial.println(value); 
} 

void setup(){ 
    Serial.begin(9600); 
    while(!Serial); 
    Serial.println("Serial ready @ 9600 bauds"); 
    // attempt to connect to Wifi network: 
    while ( status != WL_CONNECTED) { 
        status = WiFi.begin(ssid, keyIndex, pass); 
    } 
    Serial.println("Connected to wifi !"); 
    while(!webgui.connect(&client, server)); //try to connect as many times as needed
    Serial.println("Connected to webgui server !"); 
    webgui.reset(); 
    webgui.addInputString("Enter your name", &actionString); 
} 

void loop(){ 
    webgui.update(); 
}
