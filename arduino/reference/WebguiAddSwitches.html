addSwitches()

Description

Adds a control. This control is a list of switches. The user can switch on/off the switches, and then the callback function will be executed with the actual value of the switches as a parameter.

Syntax

webgui.addSwitches("Name of the switches", numswitches, switches, fnAction)

Headers

int addSwitches(const char * name, unsigned char numswitches, bool * switches, CallbackTypeBoolp fnAction);

Parameters

name: The name of the switches list

numswitches: The number of switches.

switches: Array of bool witch each element of the array is the default value of the switches. The length of this array must be the same as the numswitches value.

fnAction: Pointer to function that takes an array of bool as a parameter and returns void. This parameter is the actual status of the switches. This function will be called each time the user changes the value of any switch within the control.

Returns

The int id of the element

Example

#include <Webgui.h> 
#include <WiFi.h> 
  
Webgui webgui; //initialize an instance of the class 

char ssid[] = "Mywifi"; //  your network SSID (name)  
char pass[] = "0123ABCDEF";    // your network password (use for WPA, or use as key for WEP) 
int keyIndex = 0;// your network key Index number (needed only for WEP) 
char server[] = "192.168.1.17"; 
WiFiClient client; 
int status = WL_IDLE_STATUS; 

int ledspins[] = {6, 8, 9, 10};


void actionBools(bool * value){ 
    Serial.print("Leds status: "); 
    for (int i = 0; i < 4; ++i){ 
        Serial.print(value[i] ? "ON " : "OFF "); 
        digitalWrite(ledspins[i], (value[i] ? HIGH : LOW)); 
    } 
    Serial.println(); 
} 

void setup(){ 
    for (int i = 0; i < 4; ++i){ 
        pinMode(ledspins[i], OUTPUT); 
    } 
    Serial.begin(9600); 
    while(!Serial); 
    Serial.println("Serial ready @ 9600 bauds"); 
    // attempt to connect to Wifi network: 
    while ( status != WL_CONNECTED) { 
        status = WiFi.begin(ssid, keyIndex, pass); 
    } 
    Serial.println("Connected to wifi !"); 
    while(!webgui.connect(&client, server)); //try to connect as many times as needed
    Serial.println("Connected to webgui server !"); 
    webgui.reset(); 
    bool sws[] = {false, false, false, false};
    webgui.addSwitches("Leds", 4, sws, &actionBools); 
    actionBools(sws); 
} 

void loop(){ 
    webgui.update(); 
    delay(100); 
}
